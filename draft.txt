const
   BASE_URL = import.meta.env.VITE_REST_BASE_URL;

export async function apiFetch<T>(
  path: string,
  options: RequestInit = {}
): Promise<T> {
  const response = await fetch(`${BASE_URL}${path}`, {
    headers: {
      "Content-Type": "application/json",
      ...(options.headers || {}),
    },
    ...options,
  });

  if (!response.ok) {
    const errorBody = await response.json().catch(() => null);
    throw new Error(errorBody?.message || "API request failed");
  }

  return response.json();
}

  import { apiFetch } from "./client";

export type Event = {
  id: number;
  title: string;
  description: string;
  typeId: number;
  ownerId: number;
};

export function getEvents(): Promise<Event[]> {
  return apiFetch<Event[]>("/api/events");
}

export function getEventById(eventId: number): Promise<Event> {
  return apiFetch<Event>(`/api/events/${eventId}`);
}

export function createEvent(request: {
  title: string;
  description: string;
  typeId: number;
  ownerId: number;
}): Promise<Event> {
  return apiFetch<Event>("/api/events", {
    method: "POST",
    body: JSON.stringify(request),
  });
}
